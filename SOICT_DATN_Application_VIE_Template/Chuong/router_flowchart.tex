\begin{figure}[htbp]
\centering
\resizebox{\textwidth}{!}{%
\begin{tikzpicture}[
node distance=0.6cm and 0.6cm,
>=Stealth,
box/.style={
  draw, rounded corners=3pt, align=center,
  font=\small, minimum width=2.8cm, minimum height=0.75cm,
  fill=gray!6
},
input/.style={box, fill=green!10},
process/.style={box, fill=blue!8},
decision/.style={box, fill=orange!12, diamond, aspect=2},
output/.style={box, fill=green!10},
reject/.style={box, fill=red!8},
detail/.style={box, fill=yellow!8, font=\tiny, minimum width=4.2cm},
line/.style={->, thick},
dline/.style={->, thick, dashed, gray!50}
]

% ══════════════════════════════════════════════════════════════════
% INPUT
% ══════════════════════════════════════════════════════════════════
\node[input] (inp) {
  \textbf{Đầu vào:} Crop biển số\\
  sau nắn chỉnh phối cảnh
};

% ══════════════════════════════════════════════════════════════════
% QUALITY ROUTER (classifier + heuristic)
% ══════════════════════════════════════════════════════════════════
\node[process, below=0.9cm of inp, fill=blue!10, minimum width=3.5cm] (qr) {
  \textbf{Quality Router}\\
  Phân loại + Chẩn đoán suy giảm
};

% ══════════════════════════════════════════════════════════════════
% 3 ROUTING DECISIONS
% ══════════════════════════════════════════════════════════════════
% Decision: legibility
\node[decision, below=0.8cm of qr, aspect=2, minimum width=2cm] (leg) {Mức độ\\đọc được?};

% Three branches
\node[process, below=1.0cm of leg, fill=green!10] (direct) {
  \textbf{perfect / good}\\
  \textit{direct}
};

\node[process, left=1.5cm of direct, fill=orange!8] (fusion) {
  \textbf{poor}\\
  \textit{tracklet\_fusion}
};

\node[reject, right=1.5cm of direct] (illeg) {
  \textbf{illegible}\\
  \textit{unreadable\_wait}
};

% ══════════════════════════════════════════════════════════════════
% DIRECT PATH → OCR ngay
% ══════════════════════════════════════════════════════════════════
\node[process, below=0.7cm of direct, fill=blue!8] (ocr_direct) {
  OCR trực tiếp\\
  SmallLPR-Line-CTC
};

\node[decision, below=0.6cm of ocr_direct, aspect=2, minimum width=2.2cm] (conf_ok) {
  Đạt ngưỡng?\\
  (conf $\ge$ 0.9)
};

\node[process, below=0.7cm of conf_ok, fill=green!10] (accept) {
  Chấp nhận kết quả\\
  cập nhật track
};

\node[process, below=0.7cm of accept, fill=gray!10] (buf_direct) {
  Lưu vào track buffer\\
  chờ tổng hợp sau
};

% ══════════════════════════════════════════════════════════════════
% FUSION PATH → Buffer + Enhancement + OCR
% ══════════════════════════════════════════════════════════════════
\node[process, below=0.7cm of fusion, fill=orange!8] (buf_fusion) {
  Lưu vào track buffer\\
  (chờ đủ bằng chứng)
};

% Enhancement pipeline
\node[process, below=0.6cm of buf_fusion, fill=yellow!10] (enhance) {
  Tăng cường ảnh\\
  (7 kỹ thuật)
};

\node[process, below=0.6cm of enhance, fill=blue!8] (ocr_fusion) {
  OCR trên ảnh\\
  tăng cường
};

\node[process, below=0.6cm of ocr_fusion, fill=blue!8] (rerank) {
  Xếp hạng lại kết quả\\
  (rerank)
};

% ══════════════════════════════════════════════════════════════════
% UNREADABLE PATH → Skip OCR
% ══════════════════════════════════════════════════════════════════
\node[process, below=0.7cm of illeg, fill=red!8] (skip) {
  Lưu làm bằng chứng\\
  không chạy OCR
};

% ══════════════════════════════════════════════════════════════════
% CONVERGENCE → Track end → CTM Voting
% ══════════════════════════════════════════════════════════════════
\node[process, below=2.5cm of conf_ok, fill=purple!8, minimum width=4cm] (track_end) {
  \textbf{Track kết thúc / Video xử lý xong}
};

\node[process, below=0.6cm of track_end, fill=blue!10, minimum width=4cm] (ctm) {
  \textbf{Tổng hợp CTM:} Bỏ phiếu theo vị trí\\
  (35 mẫu VN, quy hoạch động, vote by slot)
};

\node[output, below=0.6cm of ctm, minimum width=4cm] (final) {
  \textbf{Kết quả cuối cùng:}\\
  Chuỗi biển số + độ tin cậy\\
  + ảnh crop tốt nhất + ảnh phương tiện
};

% ══════════════════════════════════════════════════════════════════
% DETAIL BOXES (bên phải)
% ══════════════════════════════════════════════════════════════════

% QR detail
\node[detail, right=0.15cm of qr, minimum width=5.0cm] (d_qr) {
  \textbf{Hai đầu ra:}\\
  $\bullet$ Nhãn 4 lớp: perfect / good / poor / illegible\\
  $\bullet$ Điểm chất lượng $q \in [0,1]$:\\
  \quad $q = 0.8 \times$ sắc nét (Laplacian) $+\ 0.2 \times$ kích thước\\
  $\bullet$ 7 dấu hiệu suy giảm: mờ, thấp sáng, thấp tương phản,\\
  \quad mưa/sương, lỗi màu, độ phân giải thấp, che khuất
};

% Direct detail
\node[detail, right=0.15cm of ocr_direct, minimum width=4.5cm] (d_direct) {
  Nếu OCR đạt ngưỡng 0.90\\
  và định dạng VN hợp lệ\\
  $\rightarrow$ cập nhật track ngay\\
  Nếu không $\rightarrow$ chuyển sang buffer\\
  để tổng hợp đa khung
};

% Enhancement detail
\node[detail, right=0.15cm of enhance, minimum width=4.5cm] (d_enh) {
  \textbf{7 kỹ thuật tăng cường (tùy theo suy giảm):}\\
  $\bullet$ sharpen: làm sắc nét (mờ chuyển động, độ phân giải thấp)\\
  $\bullet$ gamma: điều chỉnh độ sáng (thiếu sáng)\\
  $\bullet$ CLAHE: tăng tương phản (tương phản thấp)\\
  $\bullet$ contrast\_stretch: giãn tương phản\\
  $\bullet$ denoise: lọc nhiễu (mưa/sương)\\
  $\bullet$ haze\_contrast: CLAHE + stretch kết hợp\\
  $\bullet$ white\_balance: cân bằng trắng (lỗi màu)
};

% Rerank detail
\node[detail, right=0.15cm of rerank, minimum width=4.5cm] (d_rerank) {
  \textbf{Điểm xếp hạng mỗi ứng viên:}\\
  $=$ độ tin cậy TB\\
  $+\ 0.35$ nếu định dạng hợp lệ\\
  $-\ 0.25$ nếu không hợp lệ\\
  $+\ 0.04$ nếu phương pháp gốc\\
  $-$ phạt rủi ro\\
  $+\ 0.06 \times$ số lần xuất hiện trong lịch sử track
};

% CTM detail
\node[detail, right=0.15cm of ctm, minimum width=4.5cm, fill=purple!5] (d_ctm) {
  \textbf{Đóng góp CTM:}\\
  Vote theo từng vị trí thay vì cả chuỗi\\
  $\rightarrow$ giữ ký tự đúng ngay cả khi\\
  \quad một số vị trí khác nhiễu\\
  \textbf{Đóng góp Segment Voting:}\\
  Tách biển thành (tỉnh, seri, số)\\
  vote riêng từng phần rồi ghép lại
};

% ══════════════════════════════════════════════════════════════════
% ARROWS — Main flow
% ══════════════════════════════════════════════════════════════════
\draw[line] (inp) -- (qr);
\draw[line] (qr) -- (leg);

% Legibility branches
\draw[line] (leg) -- node[left, font=\small] {perfect / good} (direct);
\draw[line] (leg) -- node[above left, font=\small] {poor} (fusion);
\draw[line] (leg) -- node[above right, font=\small] {illegible} (illeg);

% Direct path
\draw[line] (direct) -- (ocr_direct);
\draw[line] (ocr_direct) -- (conf_ok);
\draw[line, green!60!black] (conf_ok) -- node[right, font=\small] {Có} (accept);
\draw[line] (accept) -- (buf_direct);

% Direct → fallback to buffer
\draw[line, orange!60] (conf_ok) -- node[above, font=\tiny, sloped] {Không} ++(1.5,0) |- ++(0,-1.2) coordinate (fallback) -- (buf_direct);

% Fusion path
\draw[line] (fusion) -- (buf_fusion);
\draw[line] (buf_fusion) -- (enhance);
\draw[line] (enhance) -- (ocr_fusion);
\draw[line] (ocr_fusion) -- (rerank);

% Unreadable path
\draw[line] (illeg) -- (skip);

% All paths converge to track end
\draw[line] (buf_direct) -- ++(0,-0.8) -| (track_end);
\draw[line] (rerank) -- ++(0,-1.8) -| (track_end);
\draw[line] (skip) -- ++(0,-0.5) -| node[pos=0.3, right, font=\tiny] {bằng chứng} (track_end);

% Track end → CTM → Final
\draw[line, line width=1.2pt] (track_end) -- (ctm);
\draw[line, line width=1.2pt] (ctm) -- (final);

% ══════════════════════════════════════════════════════════════════
% DETAIL CONNECTIONS (dashed)
% ══════════════════════════════════════════════════════════════════
\draw[dline] (qr.east) -- (d_qr.west);
\draw[dline] (ocr_direct.east) -- (d_direct.west);
\draw[dline] (enhance.east) -- (d_enh.west);
\draw[dline] (rerank.east) -- (d_rerank.west);
\draw[dline] (ctm.east) -- (d_ctm.west);

% ══════════════════════════════════════════════════════════════════
% LEGEND
% ══════════════════════════════════════════════════════════════════
\node[below=0.3cm of final, font=\tiny, align=left] {
  \textbf{Đóng góp của hệ thống:}\\
  $\bullet$ Quality Router: định tuyến thông minh theo chất lượng ảnh\\
  $\bullet$ 7 kỹ thuật tăng cường ảnh tùy theo loại suy giảm\\
  $\bullet$ CTM Voting: bỏ phiếu theo vị trí (không phải cả chuỗi)\\
  $\bullet$ Segment Voting: tách biển thành từng phần trước khi vote
};

\end{tikzpicture}%
}
\caption{Luồng xử lý sau Quality Router — định tuyến và tổng hợp đa khung hình}
